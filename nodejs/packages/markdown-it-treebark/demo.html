Template with Data

```treebark
$template:
  div:
    class: product-grid
    $children:
      - h3: "Products"
      - div:
          class: products
          $bind: items
          $children:
            - div:
                class: product-card
                $children:
                  - h4: "{{name}}"
                  - p: "{{desc}}"
                  - div:
                      class: price
                      $children: ["{{price}}"]
$data:
  items:
    - name: "Core"
      desc: "Basic features"
      price: "Free"
    - name: "Pro"
      desc: "Advanced features"
      price: "$29/mo"
```
